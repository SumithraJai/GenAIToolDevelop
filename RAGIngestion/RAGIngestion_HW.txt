```java
package week1.day2;

import java.util.HashSet;
import java.util.Set;

public class PrintDuplicatesInArray {

    /**
     * Prints duplicate elements in the input array.
     * 
     * @param inputArray the input array
     */
    public static void printDuplicates(int[] inputArray) {
        Set<Integer> uniqueElements = new HashSet<>();
        Set<Integer> duplicateElements = new HashSet<>();

        for (int element : inputArray) {
            if (!uniqueElements.add(element)) {
                duplicateElements.add(element);
            }
        }

        for (Integer duplicate : duplicateElements) {
            System.out.println("Duplicate value: " + duplicate);
        }
    }

    public static void main(String[] args) {
        int[] inputArray = {14, 12, 13, 11, 15, 14, 18, 16, 17, 19, 18, 17, 20, 12};
        printDuplicates(inputArray);
    }
}
```
Note: The improved code uses a HashSet to store unique elements and a separate HashSet to store duplicate elements. The time complexity is reduced to O(n), and the code is more modular and maintainable.
